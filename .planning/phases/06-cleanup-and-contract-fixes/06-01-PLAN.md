**useBills.ts changes — add updateBill function:**
1. Add an `updateBill` callback after the existing `deleteBill` callback (around line 168). It updates the `bill_templates` row and also the corresponding `monthly_bills` row so the UI reflects changes immediately:
   ```typescript
   const updateBill = useCallback(
     async (templateId: string, updates: { name?: string; amount?: number; due_day?: number | null }) => {
       // Update the template
       const { error: templateError } = await supabase
         .from('bill_templates')
         .update(updates)
         .eq('id', templateId);

       if (templateError) {
         setError(templateError.message);
         return;
       }

       // Also update the monthly_bills row for this month so UI reflects changes
       if (monthId) {
         const monthlyUpdates: Record<string, unknown> = {};
         if (updates.name !== undefined) monthlyUpdates.name = updates.name;
         if (updates.amount !== undefined) monthlyUpdates.amount = updates.amount;
         if (updates.due_day !== undefined) monthlyUpdates.due_day = updates.due_day;

         if (Object.keys(monthlyUpdates).length > 0) {
           await supabase
             .from('monthly_bills')
             .update(monthlyUpdates)
             .eq('bill_template_id', templateId)
             .eq('month_id', monthId);
         }
       }

       refetch();
     },
     [monthId, refetch]
   );
   ```

     setEditName(bill.name);
     setEditAmount(String(bill.amount));
     setEditDueDay(bill.due_day ? String(bill.due_day) : '');
   };

   const cancelEdit = () => {
     setEditingId(null);
   };

   const saveEdit = async () => {
     if (!editingId || !onEditBill) return;
     const parsedAmount = parseFloat(editAmount);
     const parsedDay = editDueDay === '' ? null : parseInt(editDueDay, 10);
     if (!editName.trim() || isNaN(parsedAmount) || parsedAmount <= 0) return;
     if (editDueDay !== '' && (isNaN(parsedDay!) || parsedDay! < 1 || parsedDay! > 31)) return;

     await onEditBill(editingId, {
       name: editName.trim(),
       amount: parsedAmount,
       due_day: parsedDay,
     });
     setEditingId(null);
   };
